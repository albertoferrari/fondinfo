Iterazione: for

  • Condizione di permanenza nel ciclo
  • Controllo preliminare
  • Possibile che il corpo non sia mai eseguito
// Sum of the numbers from 1 to n
total := 0
count := 1
n := ParseInt(Prompt("n? "))

for count <= n {
    total = total + count
    count = count + 1
}
Println("The sum is ", total)
